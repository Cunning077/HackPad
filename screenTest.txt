##include <Arduino.h>
#include <SPI.h>
#include <Adafruit_GFX.h>
#include <Adafruit_GC9A01A.h>

// Shared TFT pins
#define TFT_DC   9
#define TFT_RST  8

// Since the 74HC138 is handling CS selection,
// we can use any unused pin here.
#define TFT_CS   10

Adafruit_GC9A01A tft(TFT_CS, TFT_DC, TFT_RST);

// 74HC138 select pins
#define SEL_A 7
#define SEL_B 6
define SEL_C 5

void selectDisplay(uint8_t display)
{
    digitalWrite(SEL_A, display & 0x01);
    digitalWrite(SEL_B, (display >> 1) & 0x01);
    digitalWrite(SEL_C, (display >> 2) & 0x01);

    // Allow the decoder output to settle
    delay(1);
}

void setup()
{
    pinMode(SEL_A, OUTPUT);
    pinMode(SEL_B, OUTPUT);
    pinMode(SEL_C, OUTPUT);

    // Initialize display 0
    selectDisplay(0);
    tft.begin();
    tft.setRotation(1);

    // Initialize display 1
    selectDisplay(1);
    tft.begin();
    tft.setRotation(1);
}

void loop()
{
    // Display 0
    selectDisplay(0);
    tft.fillScreen(GC9A01A_RED);
    tft.setTextColor(GC9A01A_WHITE);
    tft.setTextSize(2);
    tft.setCursor(20, 20);
    tft.print("DISPLAY 0");

    // Display 1
    selectDisplay(1);
    tft.fillScreen(GC9A01A_BLUE);
    tft.setTextColor(GC9A01A_WHITE);
    tft.setTextSize(2);
    tft.setCursor(20, 20);
    tft.print("DISPLAY 1");

    delay(1000);

    selectDisplay(0);
    tft.fillScreen(GC9A01A_GREEN);

    selectDisplay(1);
    tft.fillScreen(GC9A01A_YELLOW);

    delay(1000);
}
